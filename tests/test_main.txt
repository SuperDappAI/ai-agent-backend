import pytest
import requests
import os

@pytest.fixture
def url():
    return "http://localhost:8000" # Modify as needed

def test_query_plan(url):
    test_api_key = os.getenv("OPENAI_API_KEY")
    mock_data = {"api_key": f"{test_api_key}", "query": "Compare the temperature in Sydney to that of London today"}
    response = requests.post(f"{url}/query_plan/", json=mock_data)
    assert response.status_code == 200
    assert "response" in response.json()
    assert "elapsed_time" in response.json()
